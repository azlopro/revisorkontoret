{{- /* Responsivt billede med webp + fallback. Params: src, alt, class, loading (lazy|eager), sizes, widths (slice), fill (fx "384x384") */ -}} {{- $img := resources.Get (printf "billeder/%s" .src) }} {{- if not $img }}{{ errorf "Billede ikke fundet: %s" .src }}{{ end }} {{- $alt := .alt }} {{- $class := .class | default "" }} {{- $loading := .loading | default "lazy" }} {{- $sizes := .sizes | default "100vw" }} {{- if .fill }} {{- $webp := $img.Fill (printf "%s webp Top" .fill) }} {{- $fallback := $img.Fill (printf "%s Top" .fill) }} {{- else }} {{- $widths := .widths | default (slice 480 800 1200) }} {{- $kandidater := slice }} {{- range $w := $widths }}{{ if le $w $img.Width }}{{ $kandidater = $kandidater | append $w }}{{ end }}{{ end }} {{- if not $kandidater }}{{ $kandidater = slice $img.Width }}{{ end }} {{- $webpSet := slice }} {{- range $w := $kandidater }} {{- $r := $img.Resize (printf "%dx webp" $w) }} {{- $webpSet = $webpSet | append (printf "%s %dw" $r.RelPermalink $w) }} {{- end }} {{- $fallback := $img.Resize (printf "%dx" (index $kandidater (sub (len $kandidater) 1))) }} {{- end }}
